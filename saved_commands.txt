# Download Miniconda installer
wget https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh
# Make installer executable
chmod +x Miniconda3-latest-Linux-x86_64.sh
#Run the installer
bash Miniconda3-latest-Linux-x86_64.sh
conda --version
# =========================================
# Step 1: Activate your base conda environment
# =========================================
echo "Activating base conda environment..."
conda activate base && echo "Base conda environment activated"
# Step 2: Create a conda environment named funtools
# =========================================
echo "Creating environment 'funtools' (with Python 3.11)..."
conda create -n funtools python=3.11 -y && echo "'funtools' created"
# Step 3: Activate the funtools environment
echo "Activating 'funtools' ..."
conda activate funtools && echo "'funtools' is active"
conda activate funtools
# Step 4: Install Figlet using apt-get (or conda if preferred)
echo "Installing Figlet..."
sudo apt-get update
sudo apt-get install figlet -y
echo "Figlet installation completed"
# Step 5: Run Figlet to display your name
echo "Displaying my name with Figlet:"
figlet "Marvellous Badmus"
# Step 6: Install bwa through the bioconda channel
conda install -c bioconda bwa -y
echo "BWA installed"
# Step 7: Install blast through the bioconda channel
conda install -c bioconda blast -y
echo "BLAST installed"
# Step 8: Install samtools through the bioconda channel
conda install -c bioconda samtools -y
echo "Samtools installed"
# Step 9: Install bedtools through the bioconda channel
conda install -c bioconda bedtools -y
echo "Bedtools installed"
# Step 10: Install spades.py through the bioconda channel
conda install -c bioconda spades -y
echo "SPAdes installed"
# Step 11: Install bcftools through the bioconda channel
conda install -c bioconda bcftools -y
echo "BCFtools installed"
# Step 12: Install fastp through the bioconda channel
conda install -c bioconda fastp -y
echo "Fastp installed"
# Step 13: Install multiqc through the bioconda channel
conda install -c bioconda multiqc -y
echo "MultiQC installed"
